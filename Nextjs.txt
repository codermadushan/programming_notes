---------- Authorization (authz.ts) --------------------------------------------

export const USER_ROLES = ["admin", "user"] as const;

type AuthzRole = (typeof USER_ROLES)[number];

type AuthzAction = "create" | "read" | "update" | "delete";

type AuthzResource = "post";

type AuthzPermission = `${AuthzAction}:${AuthzResource}`;

const PERMISSIONS: Record<AuthzRole, AuthzPermission[]> = {
  admin: [
    // post
    "create:post",
    "read:post",
    "update:post",
    "delete:post",
  ],

  user: [
    // post
    "read:post",
  ],
};

type CheckPermissionProps = {
  role: AuthzRole;
  action: AuthzAction;
  resource: AuthzResource;
};

export function checkPermission({
  role,
  action,
  resource,
}: CheckPermissionProps): boolean {
  const permissions = PERMISSIONS[role];

  return permissions.includes(`${action}:${resource}`);
}

    To wrap the component which triggers the dialog.

"use client";

import { startTransition, useActionState, useEffect } from "react";

import { zodResolver } from "@hookform/resolvers/zod";
import { useForm } from "react-hook-form";
import { z } from "zod";

import { login } from "@/actions/login";
import { LoginFormSchema } from "@/lib/validators";
import {
  Form,
  FormControl,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
} from "@/components/ui/form";
import { Input } from "@/components/ui/input";
import { ButtonLoading } from "../button-loading";
import FormErrorMessage from "../form-error-message";
import showToast from "../show-toast";

export function LoginForm() {
  const [state, loginAction, isPending] = useActionState(login, undefined);

  useEffect(() => {
    if (state && state.success) {
      showToast({ success: true, message: state.message });
    } else if (state && !state.success) {
      showToast({ success: false, message: state.message });
    }
  }, [state]);

  const form = useForm<z.infer<typeof LoginFormSchema>>({
    resolver: zodResolver(LoginFormSchema),
    defaultValues: {
      email: "",
    },
  });

  function onSubmit(data: z.infer<typeof LoginFormSchema>) {
    startTransition(() => {
      const formData = new FormData();
      Object.entries(data).forEach(([key, value]) => {
        formData.append(key, value);
      });
      loginAction(formData);
    });
  }

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
        <FormField
          control={form.control}
          name="email"
          render={({ field }) => (
            <FormItem>
              <FormLabel htmlFor="email">Email</FormLabel>
              <FormControl>
                <Input
                  id="email"
                  autoComplete="off"
                  placeholder="john@example.com"
                  {...field}
                />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />

        <ButtonPending isPending={isPending}>Login</ButtonPending>
      </form>
    </Form>
  );
}

type Props = {
  className?: string;
  onSearch?: (query: string) => void;
  placeholder?: string;
};

export default function Searchbar({ className, onSearch, placeholder }: Props) {
  const [query, setQuery] = useState("");

  function handleChange(event: ChangeEvent<HTMLInputElement>) {
    setQuery(event.target.value);
  }

  function handleKeyDown(event: React.KeyboardEvent<HTMLInputElement>) {
    if (event.key === "Enter" && query) {
      onSearch?.(query);
    }
  }

  return (
    <div className="relative">
      <Search className="absolute size-4 top-1/2 -translate-y-1/2 left-2 opacity-50" />
      <Input
        id="search"
        value={query}
        className={cn("w-60 pl-8", className)}
        placeholder={placeholder ?? "Search..."}
        autoComplete="off"
        onChange={handleChange}
        onKeyDown={handleKeyDown}
      />
    </div>
  );
}

import { useMedia } from "react-use";

import NavigationDesktop from "./navigation-desktop";
import NavigationMobile from "./navigation-mobile";

const routes = [
  { href: "/", label: "Dashboard" },
  { href: "/transactions", label: "Transactions" },
  { href: "/accounts", label: "Accounts" },
  { href: "/categories", label: "Categories" },
] as const;

export default function Navigation() {
  const isMobile = useMedia("(max-width: 1024px)", false);

  if (isMobile) {
    return <NavigationMobile routes={routes} />; // Use shadcn Sheet component
  }

  return <NavigationDesktop routes={routes} />;
}

---------- Redux Toolkit -------------------------------------------------------

* Installation: (can find from official redux toolkit web page quick start)

    npm install @reduxjs/toolkit react-redux

/src
  /redux
    /features

* we should use serializable values with redux toolkit.

Primitives:

  string
  number
  boolean
  null
  undefined


Arrays of primitives:

  [1, 2, 3]
  ["a", "b"]
  [false, true]


Objects with primitive keys.

  {
    value: 1,
    isTrue: false,
    count: null,
  }




---------- (Setup) ----------

1. Create a global provider (AppProvider) for all providers (redux + other).

"use client";

import { Provider } from "react-redux";
import store from "./redux/store";

type Props = {
  children: React.ReactNode;
};

export default function AppProvider({ children }: Props) {
  return <Provider store={store}>{children}</Provider>;
}




---------- (Create store) ----------

* File name: store.ts


Ex:

import { configureStore } from "@reduxjs/toolkit";
import { TypedUseSelectorHook, useDispatch, useSelector } from "react-redux";

import counterSlice from "./features/counter-slice";

const store = configureStore({
  reducer: {
    counter: counterSlice.reducer,
  },
});

export const useAppSelector: TypedUseSelectorHook<
  ReturnType<typeof store.getState>
> = useSelector;
export const useAppDispatch: () => typeof store.dispatch = useDispatch;

export default store;




---------- (Create a slice) ----------

* File naming convension: <feature_name>-slice.ts (Ex: posts-slice.ts)


Ex:

import { createSlice, PayloadAction } from "@reduxjs/toolkit";

const initialState = {
  count: 0,
};

const counterSlice = createSlice({
  name: "counter",
  initialState,
  reducers: {
    incrementCount: (state, action: PayloadAction<number>) => {
      state.count += action.payload;
    },

    decrementCount: (state) => {
      state.count--;
    },
  },
});

export const { incrementCount, decrementCount } = counterSlice.actions;

export default counterSlice;




---------- (Listening to value changes) ----------

const value = useAppSelector((state) => state.counter.count);


* Can use only inside a client component.

* It rebuilds the component when the state is changed.
  In this case the component is listening only to count value but not other
  properties in counter state.




---------- (Just access the value without listening) ----------

const currentState = store.getState();
const value = currentState.counter.count;




---------- (Change state) ----------

const dispatch = useAppDispatch();
dispatch(decrementCount(<args_if_available>));


* Can use only inside a client component.

* It doesn't rebuild the component.




---------- Async actions ----------

* If the operation is an async, we have to use "createAsyncThunk".

1. Create the asyncThunk function as follow,

      export const incrementCount = createAsyncThunk<
        number, // success type
        number, // function args
        { rejectValue: string } // reject value
      >("counter/incrementCount", async (amount, { rejectWithValue }) => {
        try {
          await new Promise((resolve) => setTimeout(resolve, 5000));
          return amount;
        } catch (error) {
          const message =
            error instanceof Error ? error.message : "Something went wrong!";
          return rejectWithValue(message);
        }
      });


2. Now add isPending & message props to initialState.

      type InitialState = {
        count: number;
        isPending: boolean;
        message: string;
      };

      const initialState: InitialState = {
        count: 0,
        isPending: false,
        message: "",
      };


3. Now add extraReducers as follows in createSlice.

      extraReducers: (builder) => {
      builder
        .addCase(incrementCount.pending, (state) => {
          state.isPending = true;
        })
        .addCase(incrementCount.fulfilled, (state, action) => {
          state.isPending = false;
          state.count += action.payload;
          state.message = "Success";
        })
        .addCase(incrementCount.rejected, (state, action) => {
          state.isPending = false;
          state.message = action.payload ?? "Unknown error";
          console.log(state.message);
        });
      },


* Usage:
    Now you can call asyncThunk function by importing it just like exported
    action.

        dispatch(<async_thunk_function>(<args>))




---------- RTK Query ----------

* File naming convension: <feature_name>-api.ts (Ex: posts-api.ts)


* Fetching (GET):

    import { createApi, fetchBaseQuery } from "@reduxjs/toolkit/query/react"; // must be query/react

    type Post = {
      userId: number;
      id: number;
      title: string;
      body: string;
    };

    const postsApi = createApi({
      reducerPath: "posts",
      baseQuery: fetchBaseQuery({
        baseUrl: "https://jsonplaceholder.typicode.com",
      }),
      endpoints: (builder) => ({
        getPosts: builder.query<Post[], void>({
          //  query generics: result value, query callback arg
          query: () => "/posts",
        }),
      }),
    });

    export const { useGetPostsQuery } = postsApi;

    export default postsApi;


    * Access data:

      const { data: posts, isLoading, isError } = useGetPostsQuery(<args>);

          Note:
              1. During the initial render:

                  * isLoading: true

                  * data: undefined


              2. If the response status code is not 2XX (200-299), then RTK
                 Query consider it as an error.


              3. We can catch http errors as follows:

                  const { isLoading, isError, error } = useVerifyTokenQuery();

                  useEffect(() => {
                    if (isHttpError(error)) {
                      const err = error as HttpError;
                      toast.error(err.data.message);
                    } else if (error) {
                      toast.error("Something went wrong!, Please try again later");
                    }
                  }, [error]);


                  Place in src/lib/utils.ts

                      export function isHttpError(error: unknown): boolean {
                        const result =
                          error &&
                          typeof error === "object" &&
                          "status" in error &&
                          "data" in error &&
                          typeof error.status === "number";

                        return result as boolean;
                      }


                  Place in src/types/http-error.ts

                      export type HttpError = {
                        status: number;
                        data: {
                          success: boolean;
                          message: string;
                        };
                      };




* Mutations (POST, DELETE, PATCH):

    type Post = {
      userId: number;
      id: number;
      title: string;
      body: string;
    };

    const postsApi = createApi({
      reducerPath: "posts",
      baseQuery: fetchBaseQuery({
        baseUrl: "https://jsonplaceholder.typicode.com",
      }),
      endpoints: (builder) => ({
        getPosts: builder.query<Post[], void>({
          //  query generics: result value, query callback arg
          query: () => "/posts",
        }),

        createPost: builder.mutation<Post, Omit<Post, "id">>({
          query: (post) => ({
            url: "/posts",
            method: "POST",
            body: post,
          }),
        }),
      }),
    });

    export const { useCreatePostMutation, useGetPostsQuery } = postsApi;

    export default postsApi;


    * Access data:

      const [createPostMutation, { data: post, isLoading, isError }] = useCreatePostMutation();

          Note:
              During the initial render:

                  * isLoading: true

                  * data: undefined


      createPostMutation({ body: "", title: "", userId: 1 });




* Define the api slice inside the store:

    * We have to specify it in 2 places:

        1. in "reducer"

        2. in "middleware"


Ex:

export const store = configureStore({
  reducer: {
    [<apiSlice1>.reducerPath]: <apiSlice1>.reducer,
    [<apiSlice2>.reducerPath]: <apiSlice2>.reducer,
  },

  middleware: (getDefaultMiddleware) => {
    return getDefaultMiddleware().concat(
      <apiSlice1>.middleware,
      <apiSlice2>.middleware,
    );
  }
});
